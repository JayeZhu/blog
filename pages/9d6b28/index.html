TS原始数据类型

提示

原始数据类型是 TypeScript 类型系统的基础，理解它们的特性和用法是编写高质量代码的关键第一步。

# 字符串类型（string）

字符串类型用于表示文本数据，支持多种字符串操作。

// 基本字符串类型
let str: string = 'sample';

// 模版字符串类型推断
const userName = 'JayeZhu';
const sayHello = `Hello, ${userName}`; // TypeScript 推断为 string 类型

// 字符串字面量类型
type PhoneBrand = 'Apple' | 'XiaoMi' | 'HuaWei';
let myPhoneBrand: PhoneBrand = 'XiaoMi';

// 联合字符串字面量
type Theme = 'light' | 'dark' | 'system';
type Size = 'small' | 'medium' | 'large' | 'x-large';

// 复杂的子串串模式匹配 （template literal types）
type EventName = 'click' | 'mouseup' | 'mousedown';
// 生成 'onClick' | 'onMouseup' | 'onMousedown'
type HandleName = `on${Capitalize<EventName>}`

// 字符串类型的方法支持
const text: string = 'lang'
const length: number = text.length; // 4
const upperCase: string = text.toUpperCase(); // 'LANG'
const substring: string = text.substring(2); // 'lg'

# 数字类型（number）

TypeScript 中的数字类型包含所有数值，支持多种进制表示法。

// 各种进制标识
const decimal: number = 42; // 十进制
const binary: number = 0b101010; // 二进制42
const octal: number = 0o52; // 八进制42
const hexadecimal: number = 0x2A; // 十六进制

// 科学计数法
const largeNumber: number = 1.23e6; // 1230000
const smallNumber: number = 1.23e-6; // 0.00000123

// 数字分隔符
const billion: number = 1_000_000_000;
const pi: number = 3.131_592_653;

// 数字字面量类型
type StatusCode = 200 | 301 | 400 | 500;
let httpStatus = StatusCode = 200;

// 数字方法的使用
const num: number = pi;
const fixed: string = num.toFixed(2); // '3.14'
const num.isNaN(); // false

# 布尔类型（boolean）

布尔类型表示最简单的逻辑值：true或 false。

// 基本数据类型
const isActive: boolean = true;

// 布尔字面量类型
type Truthy = true;
type Falsy = false;

// 复杂布尔逻辑类型
type TriState = true | false | null | undefined;
type BinaryState = true | false

// 从其他数据类型推断到布尔类型
const name: string = 'JayeZhu'
const boolFormName: boolean = Boolean(name); // true

const emptyName: string = '';
const boolFormEmptyName: boolean = Boolean(emptyName); // false

const num: number = 0;
const boolFromNum: boolean = Boolean(num); //false

const obj: null = null;
const boolFromNull: boolean = Boolean(obj); //false

const count: undefined;
const boolFromUndefined: boolean = Boolean(num); //false

// 布尔运算的类型推断
const booleanA: boolean = true;
const booleanB: boolean = false;
const booleanAnd: boolean = booleanA && booleanB; // false
const booleanOr: boolean = booleanA || booleanB; // true

# Symbol 类型

Symbol 类型表示唯一且不可变的值，通常用作对象属性的键。

// 创建 Symbol
const uniqueKey: symbol = Symbol('description');
const anotherKey: symbol = Symbol('description');

// Symbol 是唯一的
console.log(uniqueKey === anotherKey); // false

// 作为对象属性的键
const obj = {
  [uniqueKey]: 'secret value',
  regularKey: 'normal value'
};

// 获取 Symbol 属性
console.log(obj[uniqueKey]); // 'secret value'

// 全局 Symbol 注册表
const globalSymbol = Symbol.for('global.key');
const sameGlobalSymbol = Symbol.for('global.key');
console.log(globalSymbol === sameGlobalSymbol); // true

// 常用内置 Symbols
const array = [1, 2, 3];
const iterator = array[Symbol.iterator]();

// Symbol 的属性描述
console.log(uniqueKey.toString());    // 'Symbol(description)'
console.log(Symbol.keyFor(globalSymbol)); // 'global.key'

# BigInt 类型

BigInt 类型用于表示任意精度的整数，解决了 JavaScript 中数字精度限制的问题。

// BigInt 字面量
const bigNumber: bigint = 123456789012345678901234567890n;
const bigFromNumber: bigint = BigInt(Number.MAX_SAFE_INTEGER) + 1n;

// 各种进制的 BigInt
const binaryBigInt: bigint = 0b1010101010101010101010101010101010101010101010101010101010101010n;
const hexBigInt: bigint = 0xFFFFFFFFFFFFFFFFn;

// BigInt 运算
const a: bigint = 1000000000000000000000000n;
const b: bigint = 1234567890123456789n;

const sum: bigint = a + b;
const difference: bigint = a - b;
const product: bigint = a * b;
const quotient: bigint = a / b;
const remainder: bigint = a % b;

// BigInt 与 Number 的转换
const numberValue: number = Number(bigNumber);
const stringValue: string = bigNumber.toString();

// BigInt 的比较
console.log(100n === 100);    // false: BigInt 和 Number 不严格相等
console.log(100n == 100);     // true: 值相等
console.log(100n > 50);       // true: 可以比较

# null：明确的无值

null表示一个明确的、有意赋予的空值，通常用于表示"没有对象"或"空对象引用"。

// 显式设置为 null
let emptyValue: null = null;

// 表示没有对象
let currentUser: User | null = null; // 当前没有用户登录

// 重置变量值
let data: string | null = '初始数据';
data = null; // 明确清除数据

// DOM 元素可能为 null
const element: HTMLElement | null = document.getElementById('non-existent');
if (element === null) {
    console.log('元素不存在');
}

// API 响应中的 null
interface ApiResponse<T> {
    data: T | null;    // 成功时有数据，失败时为 null
    error: string | null; // 成功时为 null，失败时有错误信息
    status: number;
}

// 数据库查询结果
function findUserById(id: number): User | null {
    // 如果找不到用户，返回 null
    return users.find(user => user.id === id) ?? null;
}

// 可选的配置项
interface AppConfig {
    theme: string;
    apiEndpoint: string | null; // 明确表示可能没有API端点
}

// 链表数据结构
interface ListNode<T> {
    value: T;
    next: ListNode<T> | null; // 链表的结束用 null 表示
}

# undefined：未定义的值

undefined表示变量已声明但尚未赋值，或者对象属性不存在。

// 变量声明但未赋值
let uninitialized: undefined;
console.log(uninitialized); // undefined

// 显式设置为 undefined
let explicitlyUndefined: undefined = undefined;

// 函数参数默认值
function greet(name: string = '陌生人'): string {
    return `你好，${name}!`;
}
console.log(greet()); // "你好，陌生人!"

// 函数没有返回值
function noReturn(): undefined {
    console.log('这个函数没有返回值');
    // 隐式返回 undefined
}

// 访问不存在的对象属性
const obj = { name: 'JayeZhu' };
const age = obj['age']; // undefined

// 可选函数参数
function createUser(name: string, age?: number): void {
    console.log(`姓名: ${name}, 年龄: ${age ?? '未提供'}`);
}
createUser('JayeZhu'); // 姓名: JayeZhu, 年龄: 未提供

// 接口中的可选属性
interface UserProfile {
    name: string;
    email?: string; // 可选属性，默认为 undefined
    phone?: string;
}

const user: UserProfile = { name: 'JayeZhu' };
console.log(user.email); // undefined
console.log(user.phone); // undefined


// 稀疏数组包含 undefined
const sparseArray: number[] = [1, , 3]; // 中间元素为 undefined
console.log(sparseArray[1]); // undefined

// 显式的 undefined 元素
const explicitUndefinedArray: (number | undefined)[] = [1, undefined, 3];
